program: declaration+ EOF
declaration: funDecl
funDecl: "fn" IDENTIFIER "(" params? ")" ("->" dataType) block
paramas: IDENTIFIER ":" dataType ("," IDENTIFIER ":" dataType)*
block: "{" statement* "}"
statement: exprStatement | varDecl
exprStatement: expr ";"
varDecl: "let" IDENTIFIER (":" dataType)? "=" expr ";"
